	\end{flalign*}

	\subsection{Listen-Konzept} % (fold)
	\label{sub:listen_konzept}
	
		Lisp-Programme sind Listen von Symbolen und Ausdrücken.

		\lstLisp[Liste in Lisp]
		\begin{lstlisting}
(1 2)
		\end{lstlisting}

		\lstHaskell[Liste in Haskell]
		\begin{lstlisting}
[1,2] == 1:2:[]
		\end{lstlisting}
